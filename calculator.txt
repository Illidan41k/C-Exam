using System;

namespace Var10_ex_bibl
{
    public class Calc
    {
        public int sum(int a, int b)
        {
            int c = a + b;
            return c;
        }
        public int min(int a,int b)
        {
            int c = a - b;
            return c;
        }
        public int mult(int a, int b)
        {
            int c = a * b;
            return c;
        }
        public int div(int a, int b)
        {
            int c = a / b;
            return c;
        }
    }
}
